In React, you can define components as class components or functional components

As class:

//
import React from 'react';

class App extends React.Component {
  render() {
    return (
      <div className="App">
        <header className="App-header">
          Hello
        </header>
      </div>
    );
  }
}

export default App;
//

As function:

//
function App() {
  return (
    <div className="App">
      <header className="App-header">
        Hello
      </header>
    </div>
  );
}

export default App;
//
